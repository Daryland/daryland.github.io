// ───────────────────────────────────────────────────────── // AlgoForge — Ichimoku Kinko Hyo Strategy // Generated by AlgoForge Strategy Builder // ───────────────────────────────────────────────────────── //@version=5 strategy("AlgoForge Ichimoku", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=5) // ── INPUTS ── tenkan_len = input.int(9, "Tenkan-sen Period", minval=1) kijun_len = input.int(26, "Kijun-sen Period", minval=1) spanb_len = input.int(52, "Senkou Span B", minval=1) disp = input.int(26, "Displacement", minval=1) rr_min = input.float(2.0, "Min R:R Ratio", step=0.5) // ── ICHIMOKU CALCULATIONS ── donchian(len) => (ta.highest(high, len) + ta.lowest(low, len)) / 2 tenkan = donchian(tenkan_len) kijun = donchian(kijun_len) spana = math.avg(tenkan, kijun) spanb = donchian(spanb_len) chikou = close[disp] // ── CLOUD BOUNDARIES (displaced forward) ── cloud_top = math.max(spana[disp], spanb[disp]) cloud_bottom = math.min(spana[disp], spanb[disp]) // ── TREND BIAS (Monthly = use security() in production) ── above_cloud = close > cloud_top below_cloud = close < cloud_bottom inside_cloud = not above_cloud and not below_cloud // ── CHIKOU CONFIRMATION ── chikou_bull = chikou > cloud_top chikou_bear = chikou < cloud_bottom // ── BASE LINE SLOPE ── kijun_slope_up = kijun > kijun[3] kijun_slope_down = kijun < kijun[3] // ── ENTRY CONDITIONS ── long_signal = above_cloud and chikou_bull // MANDATORY and kijun_slope_up // Base Line momentum and not inside_cloud // Never mid-cloud and (ta.crossover(close, kijun) or ta.crossover(close, tenkan)) short_signal = below_cloud and chikou_bear and kijun_slope_down and not inside_cloud and (ta.crossunder(close, kijun) or ta.crossunder(close, tenkan)) // ── STOP LOSS / TAKE PROFIT ── sl_long = cloud_bottom * 0.997 // 0.3% below cloud sl_short = cloud_top * 1.003 tp_long = close + (close - sl_long) * rr_min tp_short = close - (sl_short - close) * rr_min // ── ORDERS ── if long_signal strategy.entry("Long", strategy.long) strategy.exit("Long Exit", "Long", stop=sl_long, limit=tp_long) if short_signal strategy.entry("Short", strategy.short) strategy.exit("Short Exit", "Short", stop=sl_short, limit=tp_short) // ── CLOUD VISUALIZATION ── p1 = plot(spana[disp], color=color.new(color.teal, 80), linewidth=1) p2 = plot(spanb[disp], color=color.new(color.red, 80), linewidth=1) fill(p1, p2, color = spana[disp] >= spanb[disp] ? color.new(color.teal, 90) : color.new(color.red, 90)) plot(tenkan, color=color.new(color.orange, 0), linewidth=1, title="Tenkan") plot(kijun, color=color.new(color.blue, 0), linewidth=2, title="Kijun") plot(close[disp], color=color.new(color.white, 30), linewidth=1, title="Chikou")
