techniqueRecall.ts TypeScript 
interface Technique {
  id: string;
  name: string;
  position: 'Guard' | 'Mount'
    | 'Back' | 'Side Control';
  type: 'Submission' | 'Sweep' | 'Pass';
  recallScore: number; // 0–100
  lastReviewed?: Date;
  nextReview?: Date;
}

const updateRecall = (
  t: Technique,
  rating: 0 | 1 | 2 | 3 | 4 | 5
): Technique => {
  const ease = 0.1 + (rating / 5) * 0.9;
  const days = t.lastReviewed
    ? Math.round(
        (t.nextReview!.getTime() -
          t.lastReviewed.getTime()) /
          86400000 * ease
      )
    : 1;
  const next = new Date();
  next.setDate(next.getDate() + days);
  return {
    ...t,
    recallScore: Math.min(100,
      Math.round(
        t.recallScore + (rating - 2.5) * 10
      )
    ),
    lastReviewed: new Date(),
    nextReview: next,
  };
};

interface LockSession {
  habitId: string;
  startedAt: Date;
  durationMs: number;
  blockedApps: string[];
  completed: boolean;
  abandonedAt?: Date;
}

const startLock = async (
  habit: Habit
): Promise<LockSession> => {
  await BlockingManager
    .activate(habit.appsToBlock);
  const session: LockSession = {
    habitId: habit.id,
    startedAt: new Date(),
    durationMs: habit.durationMs,
    blockedApps: habit.appsToBlock,
    completed: false,
  };
  await SessionStore.save(session);
  return session;
};

const completeLock = async (
  session: LockSession
): Promise<void> => {
  await BlockingManager.deactivate();
  await StreakTracker.record(
    session.habitId
  );
  await SessionStore.update({
    ...session, completed: true,
  });
};

interface DetectionResult {
  score: number; // 0–100
  signals: Signal[];
  verdict: 'human' | 'ai' | 'uncertain';
}

interface Signal {
  name: 'perplexity' | 'burstiness'
    | 'hedging' | 'vocabulary'
    | 'structural' | 'personal_voice';
  score: number;
  weight: number;
}

const analyzeSubmission = (
  text: string
): DetectionResult => {
  const signals = runSignals(text);
  const score = signals.reduce(
    (acc, s) => acc + s.score * s.weight, 0
  );
  return {
    score,
    signals,
    verdict:
      score > 70 ? 'ai' :
      score < 30 ? 'human' : 'uncertain',
  };
};

interface ThreatEvent {
  id: string;
  timestamp: Date;
  type: 'process' | 'file'
    | 'network' | 'registry';
  severity: 'low' | 'medium'
    | 'high' | 'critical';
  threatScore: number; // 0.0–1.0
  action: 'log' | 'alert'
    | 'block' | 'quarantine';
  hash?: string; // SHA-256
}

const evaluateThreat = async (
  event: RawEvent
): Promise<ThreatEvent> => {
  const score = await mlModel
    .predict(event);
  const action =
    score > 0.9 ? 'quarantine' :
    score > 0.7 ? 'block' :
    score > 0.4 ? 'alert' : 'log';
  return {
    ...event,
    threatScore: score,
    action,
  };
};

fileScan.ts TypeScript 
interface ScanResult {
  path: string;
  sha256: string;
  threat?: ThreatInfo;
  quarantined: boolean;
  scannedAt: Date;
  source: 'local' | 'virustotal'
    | 'claude';
}

const scanFile = async (
  filePath: string
): Promise<ScanResult> => {
  const hash = await computeSHA256(filePath);
  const local = await db.signatures
    .find(hash);
  if (local) return quarantine(filePath, local);
  if (settings.virusTotalEnabled) {
    const vt = await VirusTotal.check(hash);
    if (vt.malicious > 0)
      return quarantine(filePath, vt);
  }
  return {
    path: filePath,
    sha256: hash,
    quarantined: false,
    scannedAt: new Date(),
    source: 'local',
  };
};
