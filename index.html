Recipe Vault

A mobile recipe management app to save, organize, and discover recipes. Features smart search, ingredient tracking, and custom collections for the health-conscious cook.

React Native Expo AsyncStorage Rork
recipeStore.ts TypeScript 
interface Recipe {
  id: string;
  title: string;
  ingredients: Ingredient[];
  instructions: string[];
  prepTime: number;
  servings: number;
  tags: string[];
  imageUri?: string;
}

const RecipeStore = {
  save: async (recipe: Recipe): Promise<void> => {
    const existing = await AsyncStorage
      .getItem('recipes');
    const recipes: Recipe[] = JSON.parse(
      existing ?? '[]'
    );
    const idx = recipes.findIndex(
      r => r.id === recipe.id
    );
    if (idx >= 0) recipes[idx] = recipe;
    else recipes.push(recipe);
    await AsyncStorage.setItem(
      'recipes', JSON.stringify(recipes)
    );
  },

  search: async (query: string): Promise<Recipe[]> => {
    const all = await RecipeStore.getAll();
    const q = query.toLowerCase();
    return all.filter(r =>
      r.title.toLowerCase().includes(q) ||
      r.tags.some(t => t.toLowerCase().includes(q))
    );
  },
};

interface Goal {
  id: string;
  title: string;
  milestones: Milestone[];
  targetDate: Date;
  progress: number; // 0–100
}

interface Milestone {
  id: string;
  title: string;
  completed: boolean;
  completedAt?: Date;
}

const trackProgress = (goal: Goal): Goal => {
  const completed = goal.milestones
    .filter(m => m.completed).length;
  return {
    ...goal,
    progress: goal.milestones.length
      ? Math.round(
          (completed / goal.milestones.length) * 100
        )
      : 0,
  };
};

const getNextMilestone = (goal: Goal) =>
